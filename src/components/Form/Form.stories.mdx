import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";
import { BADGE } from "@geometricpanda/storybook-addon-badges";
import { Form } from "./Form";
import { Input, Box, Button } from "components";
import { useState } from "react";
import { useForm } from "react-hook-form";

<Meta
  title="Form"
  component={Form}
  parameters={{
    badges: [BADGE.STABLE],
  }}
/>

export const Template = (args) => {
  const formHook = useForm();
  return (
    <Form {...args} width={"50%"}>
      <Box padding={"12px"}>
        <Input formInputName="email" label="Email" formHook={formHook} />
      </Box>
      <Box padding={"12px"}>
        <Input formInputName="password" label="Password" formHook={formHook} />
      </Box>
      <Box padding={"12px"} display={"flex"} justifyContent={"center"}>
        <Button
          variant="contained"
          color="primary"
          onClick={() => alert(JSON.stringify(formHook.getValues()))}
          disabled={args.isSubmitting || formHook.formState.isSubmitting}
        >
          Login
        </Button>
      </Box>
    </Form>
  );
};
