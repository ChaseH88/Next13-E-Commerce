import { useState } from "react";

<Meta
  title="Form"

  parameters={{
    badges: [BADGE.STABLE],
  }}

export const Template = (args) => {
  const formHook = useForm();
  return (
    <Form {...args}>
      <Input
        formInputName="email"
        label="Email"
        placeholder="Email"
        formHook={formHook}
      />
      <Input
        formInputName="password"
        label="Password"
        placeholder="Password"
        formHook={formHook}
      />
      <Button
        variant="primary"
        onClick={() => alert(JSON.stringify(formHook.getValues()))}
      >
        Submit
      </Button>
    </Form>
  );
};
